import aiohttp
import time
import random
from typing import List, Dict

class LoadTester:
    def __init__(self, base_url: str, concurrent_users: int = 10):
        self.base_url = base_url
        self.concurrent_users = concurrent_users
        self.results = []
        
    async def test_endpoint(self, endpoint: str, method: str = 'GET', data: Dict = None):
        """Ğ¢ĞµÑÑ‚Ğ¸Ñ€Ğ¾Ğ²Ğ°Ğ½Ğ¸Ğµ ÑĞ½Ğ´Ğ¿Ğ¾Ğ¸Ğ½Ñ‚Ğ°"""

            async with aiohttp.ClientSession() as session:
                if method == 'GET':
                    async with session.get(f"{self.base_url}{endpoint}") as response:
                        if response.status == 200:
                            success = True
                        else:
                            error = f"HTTP {response.status}"
                elif method == 'POST':
                    async with session.post(f"{self.base_url}{endpoint}", json=data) as response:
                        if response.status == 200:
                            success = True
                        else:
                            error = f"HTTP {response.status}"

            error = str(e)

        duration = end_time - start_time

            'endpoint': endpoint,
            'method': method,
            'success': success,
            'duration': duration,
            'error': error,
            'timestamp': time.time()
        }
        
        self.results.append(result)
        return result
        
    async def run_concurrent_test(self, endpoints: List[str], duration: int = 60):
        """Ğ—Ğ°Ğ¿ÑƒÑĞº Ğ¿Ğ°Ñ€Ğ°Ğ»Ğ»ĞµĞ»ÑŒĞ½Ğ¾Ğ³Ğ¾ Ñ‚ĞµÑÑ‚Ğ¸Ñ€Ğ¾Ğ²Ğ°Ğ½Ğ¸Ñ"""

        async def worker():
            while time.time() - start_time < duration:
                endpoint = random.choice(endpoints)
                await self.test_endpoint(endpoint)
                await asyncio.sleep(random.uniform(0.1, 1.0))
                
        tasks = [worker() for _ in range(self.concurrent_users)]
        await asyncio.gather(*tasks)
        
    def generate_report(self):
        """Ğ“ĞµĞ½ĞµÑ€Ğ°Ñ†Ğ¸Ñ Ğ¾Ñ‚Ñ‡ĞµÑ‚Ğ° Ğ¾ Ñ‚ĞµÑÑ‚Ğ¸Ñ€Ğ¾Ğ²Ğ°Ğ½Ğ¸Ğ¸"""

            'successful_requests': successful_requests,
            'failed_requests': failed_requests,
            'success_rate': success_rate,

        }

import re
import json

from datetime import datetime, timedelta

class LogAnalyzer:
    def __init__(self, log_file: str):
        self.log_file = log_file
        self.patterns = {
            'error': r'ERROR|error|Error',
            'warning': r'WARNING|warning|Warning',
            'task': r'Task.*?(\w+)',
            'node': r'Node.*?(\w+)',
            'performance': r'(\d+\.\d+)ms'
        }
        
    def analyze_logs(self, hours: int = 24):
        """ĞĞ½Ğ°Ğ»Ğ¸Ğ· Ğ»Ğ¾Ğ³Ğ¾Ğ² Ğ·Ğ° Ğ¿Ğ¾ÑĞ»ĞµĞ´Ğ½Ğ¸Ğµ N Ñ‡Ğ°ÑĞ¾Ğ²"""
        
        cutoff_time = datetime.now() - timedelta(hours=hours)
        results = defaultdict(list)
        
        with open(self.log_file, 'r') as f:
            for line in f:
                try:
                    timestamp_str = re.search(r'(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2})', line)
                    if timestamp_str:
                        timestamp = datetime.strptime(timestamp_str.group(1), '%Y-%m-%d %H:%M:%S')
                        if timestamp >= cutoff_time:

                except Exception as e:
                    print(f"ĞÑˆĞ¸Ğ±ĞºĞ° Ğ¾Ğ±Ñ€Ğ°Ğ±Ğ¾Ñ‚ĞºĞ¸ ÑÑ‚Ñ€Ğ¾ĞºĞ¸: {e}")
                    
        return self._generate_report(results)

            if matches:

    def _generate_report(self, results: dict):
        """Ğ“ĞµĞ½ĞµÑ€Ğ°Ñ†Ğ¸Ñ Ğ¾Ñ‚Ñ‡ĞµÑ‚Ğ°"""

            }
